window.addEventListener('DOMContentLoaded', (event) => { let backToTopButton = document.getElementById("backToTopBtn"); // console.log("Botão BackToTop encontrado:", backToTopButton); if (backToTopButton) { let scrollThreshold = 200; function toggleBackToTopButton() { if (!document.getElementById("backToTopBtn")) return; if (document.body.scrollTop > scrollThreshold || document.documentElement.scrollTop > scrollThreshold) { backToTopButton.classList.add("show"); } else { backToTopButton.classList.remove("show"); } } window.addEventListener('scroll', toggleBackToTopButton); toggleBackToTopButton(); // Verifica o estado inicial ao carregar // NENHUM OUVINTE DE CLIQUE NECESSÁRIO AQUI SE O FUNCIONAR BEM } else { // console.error("Elemento do botão BackToTop não encontrado no DOM!"); } });
